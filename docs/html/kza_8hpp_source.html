86template<class ValueT, std::unsigned_integral SizeT,
87 std::unsigned_integral WinSizeT>
88class KZGeneric
89{
90 typedef std::barrier<std::function<void()>> barrier_t;
91
92public:
97 inline void set_iterations(SizeT iterations)
98 {
99 this->iterations = iterations;
100 }
101
102protected:
108 struct ThreadData {
109 std::thread thread;
110 SizeT start_idx, end_idx;
111 };
112
114 SizeT iterations;
116 SizeT task_size;
118 SizeT threads_cnt;
119#ifdef KZA_TIMER
120 Timer timer;
121#endif
122
123
127 KZGeneric(SizeT iterations): iterations(iterations)
128 {
129 threads_cnt = std::thread::hardware_concurrency();
130#ifdef KZA_DEBUG
131 std::cout << "Number of cores: " << threads_cnt << std::endl;
132#endif
133 }
134
135 virtual ~KZGeneric() = default;
136
143 virtual void perform_single_iteration(SizeT start_idx, SizeT end_idx) = 0;
144
156 void perform_iterations(std::function<void()> on_iteration_complete)
157 {
158 barrier_t sync_iteration(threads_cnt, on_iteration_complete);
159 std::vector<ThreadData> th(threads_cnt);
160
161#ifdef KZA_TIMER
162 timer.start();
163#endif
164 this->start_threads(th, sync_iteration);
165
166 for (auto &thread_data : th)
167 thread_data.thread.join();
168
169#ifdef KZA_TIMER
170 timer.stop();
171 timer.print_elapsed();
172#endif
173 }
174
183 inline WinSizeT window_left_bound(SizeT win_center, WinSizeT half_win)
184 {
185 return (win_center >= half_win)
186 ? (win_center - half_win)
187 : 0;
188 }
189
201 inline WinSizeT window_right_bound(SizeT win_center, WinSizeT half_win,
202 SizeT data_size)
203 {
204 return (win_center + half_win >= data_size)
205 ? data_size - 1
206 : win_center + half_win;
207 }
208
209private:
216 void worker(ThreadData &t_data, barrier_t &sync_iteration)
217 {
218 for (SizeT i = 0; i < iterations - 1; ++i) {
219 this->perform_single_iteration(t_data.start_idx, t_data.end_idx);
220
221 // waiting for other threads to complete iteration
222 sync_iteration.arrive_and_wait();
223 }
224 // perform last iteration and finish
225 this->perform_single_iteration(t_data.start_idx, t_data.end_idx);
226 }
227
235 void start_threads(std::vector<ThreadData> &th, barrier_t &sync_iteration)
236 {
237 SizeT thread_task_size = (task_size + threads_cnt-1) / threads_cnt;
238 SizeT start_idx = 0;
239
240 for (auto &t_data : th) {
241 t_data.start_idx = start_idx;
242 t_data.end_idx = (start_idx + thread_task_size >= task_size)
243 ? task_size - 1
244 : start_idx + thread_task_size;
245
246 start_idx = t_data.end_idx;
247
248 t_data.thread = std::thread(&KZGeneric::worker, this,
249 std::ref(t_data),
250 std::ref(sync_iteration));
251 }
252 }
253};

259template<class ValueT, std::unsigned_integral SizeT,
260 std::unsigned_integral WinSizeT>
261class KZ1D: public KZGeneric<ValueT, SizeT, WinSizeT>
262{
263public:
268 KZ1D(WinSizeT window_size, SizeT iterations)
269 : KZGeneric<ValueT, SizeT, WinSizeT>(iterations)
270 {
271 half_window = window_size >> 1; // window half (window is 2*m+1)
272#ifdef KZA_TIMER
273 KZGeneric<ValueT, SizeT, WinSizeT>::timer.set_name("kz1d");
274#endif
275 }
276
281 std::vector<ValueT> operator()(const std::vector<ValueT> &vec)
282 {
283 task_size = vec.size();
284
285 data.assign(vec.cbegin(), vec.cend());
286 ans.resize(vec.size());
287
288#ifdef PREFIX_SUM
289 pref_sum.resize(data.size() + 1);
290 pref_finite_cnt.resize(data.size() + 1);
291 update_prefix_sum(data);
292#endif
293
294 if (iterations <= 0)
295 return std::move(data);
296
297 this->perform_iterations([this]() {
298#ifdef PREFIX_SUM
299 this->update_prefix_sum(ans);
300#else
301 std::swap(data, ans);
302#endif
303 });
304
305 return std::move(ans);
306 }
307
313 inline virtual void set_window(WinSizeT window)
314 {
315 half_window = window >> 1;
316 }
317
318protected:
319
320#ifdef PREFIX_SUM
322 std::vector<ValueT> pref_sum;
323 std::vector<SizeT> pref_finite_cnt;
324
331 void update_prefix_sum(const std::vector<ValueT> &data)
332 {
333 pref_sum[0] = 0;
334 pref_finite_cnt[0] = 0;
335 bool is_finite_flag;
336
337 for (SizeT i = 1; size_t(i) <= data.size(); ++i) {
338 is_finite_flag = std::isfinite(data[i-1]);
339 pref_sum[i] = pref_sum[i-1] + (is_finite_flag ? data[i-1] : 0);
340 pref_finite_cnt[i] = pref_finite_cnt[i-1] + is_finite_flag;
341 }
342 }
343
351 ValueT average(SizeT start_idx, SizeT end_idx)
352 {
353 /* (window sum) = (sum containig window) - (sum before window) */
354 SizeT z =
355 (pref_finite_cnt[end_idx + 1] - pref_finite_cnt[start_idx]);
356
357 return (z != 0) ? (pref_sum[end_idx + 1] - pref_sum[start_idx]) / z
358 : std::numeric_limits<ValueT>::quiet_NaN();
359 }
360
361#else
362
369 ValueT average(SizeT start_idx, SizeT end_idx)
370 {
371 SizeT z = 0;
372 ValueT s = 0;
373 bool is_finite_flag;
374
375 for (SizeT i = start_idx; i <= end_idx; ++i) {
376 is_finite_flag = std::isfinite(data[i]);
377 z += is_finite_flag;
378 s += (is_finite_flag ? data[i] : 0);
379 }
380
381 return (z != 0) ? s/z : std::numeric_limits<ValueT>::quiet_NaN();
382 }
383
384#endif
385
386private:
387
388 inline void perform_single_iteration(SizeT start_idx,
389 SizeT end_idx) override
390 {
391 for (SizeT time = start_idx; time <= end_idx; ++time)
392 ans[time] = this->average(
393 this->window_left_bound(time, half_window),
394 this->window_right_bound(time, half_window, data.size()));
395 }
396
397protected:
399 WinSizeT half_window;
401 std::vector<ValueT> data;
403 std::vector<ValueT> ans;
404
405 using KZGeneric<ValueT, SizeT, WinSizeT>::task_size;
406 using KZGeneric<ValueT, SizeT, WinSizeT>::iterations;
407 using KZGeneric<ValueT, SizeT, WinSizeT>::threads_cnt;
408};

413template<class ValueT, std::unsigned_integral SizeT,
414 std::unsigned_integral WinSizeT>
415class KZ2D: public KZGeneric<ValueT, SizeT, WinSizeT>
416{
417public:
422 KZ2D(const std::vector<WinSizeT> window_sizes, SizeT iterations)
423 : KZGeneric<ValueT, SizeT, WinSizeT>(iterations)
424 {
425 this->set_window(window_sizes);
426#ifdef KZA_TIMER
427 KZGeneric<ValueT, SizeT, WinSizeT>::timer.set_name("kz2d");
428#endif
429 }
430
435 std::vector<std::vector<ValueT>> operator()(
436 const std::vector<std::vector<ValueT>> &vec)
437 {
438 task_size = vec.size();
439
440 data = vec;
441 rows = data.size();
442 cols = data[0].size();
443 ans.resize(rows, std::vector<ValueT>(cols));
444
445#ifdef PREFIX_SUM
446 pref_sum.resize(rows + 1, std::vector<ValueT>(cols + 1, 0));
447 pref_finite_cnt.resize(rows + 1, std::vector<SizeT>(cols + 1, 0));
448 update_prefix_sum(data);
449#endif
450
451 if (iterations <= 0)
452 return std::move(data);
453
454 this->perform_iterations([this]() {
455#ifdef PREFIX_SUM
456 this->update_prefix_sum(ans);
457#else
458 data.swap(ans);
459#endif
460 });
461
462 return std::move(ans);
463 }
464
465 inline virtual void set_window(const std::vector<WinSizeT> &window)
466 {
467 half_win_rows = window[0] >> 1;
468 half_win_cols = window.size() > 1 ? window[1] >> 1 : half_win_rows;
469 }
470
471protected:
472
473#ifdef PREFIX_SUM
475 std::vector<std::vector<ValueT>> pref_sum;
476 std::vector<std::vector<SizeT>> pref_finite_cnt;
477
484 void update_prefix_sum(const std::vector<std::vector<ValueT>> &data)
485 {
486 bool is_finite_flag;
487
488 for (SizeT i = 1; i <= rows; ++i) {
489 for (SizeT j = 1; j <= cols; ++j) {
490 is_finite_flag = std::isfinite(data[i-1][j-1]);
491 pref_sum[i][j] =
492 pref_sum[i-1][j] + pref_sum[i][j-1] - pref_sum[i-1][j-1];
493 pref_sum[i][j] += (is_finite_flag ? data[i-1][j-1] : 0);
494 pref_finite_cnt[i][j] = pref_finite_cnt[i-1][j] +
495 pref_finite_cnt[i][j-1] -
496 pref_finite_cnt[i-1][j-1] +
497 is_finite_flag;
498 }
499 }
500 }
501
511 ValueT average(SizeT start_row, SizeT end_row,
512 SizeT start_col, SizeT end_col)
513 {
514 ValueT s = pref_sum[end_row+1][end_col+1] -
515 pref_sum[end_row+1][start_col] -
516 pref_sum[start_row][end_col+1] +
517 pref_sum[start_row][start_col];
518
519 SizeT z = pref_finite_cnt[end_row+1][end_col+1] -
520 pref_finite_cnt[end_row+1][start_col] -
521 pref_finite_cnt[start_row][end_col+1] +
522 pref_finite_cnt[start_row][start_col];
523
524 return (z != 0) ? s/z : std::numeric_limits<ValueT>::quiet_NaN();
525 }
526
527#else
528
537 ValueT average(SizeT start_row, SizeT end_row,
538 SizeT start_col, SizeT end_col)
539 {
540 SizeT z = 0;
541 ValueT s = 0;
542
543 for (SizeT i = start_row; i <= end_row; ++i) {
544 for (SizeT j = start_col; j <= end_col; ++j) {
545 bool is_finite_flag = std::isfinite(data[i][j]);
546 z += is_finite_flag;
547 s += (is_finite_flag ? data[i][j] : 0);
548 }
549 }
550
551 return (z != 0) ? s/z : std::numeric_limits<ValueT>::quiet_NaN();
552 }
553
554#endif
555
556private:
557
558 inline void perform_single_iteration(SizeT start_idx,
559 SizeT end_idx) override
560 {
561 for (SizeT row = start_idx; row <= end_idx; ++row) {
562 for (SizeT col = 0; col < cols; ++col) {
563 ans[row][col] = this->average(
564 this->window_left_bound(row, half_win_rows),
565 this->window_right_bound(row, half_win_rows, rows),
566 this->window_left_bound(col, half_win_cols),
567 this->window_right_bound(col, half_win_cols, cols));
568 }
569 }
570 }
571
572protected:
574 WinSizeT half_win_rows;
575
577 WinSizeT half_win_cols;
578
579 std::vector<std::vector<ValueT>> data, ans;
580 SizeT rows, cols;
581
582 using KZGeneric<ValueT, SizeT, WinSizeT>::task_size;
583 using KZGeneric<ValueT, SizeT, WinSizeT>::iterations;
584 using KZGeneric<ValueT, SizeT, WinSizeT>::threads_cnt;
585};

591template<class ValueT, std::unsigned_integral SizeT,
592 std::unsigned_integral WinSizeT>
593class KZAExtension
594{
595public:
601 inline void set_min_window(WinSizeT min_window)
602 {
603 min_window_size = min_window;
604 }
605
611 inline void set_tolerance(ValueT tol)
612 {
613 tolerance = tol;
614 }
615
623 inline SizeT adaptive_window(WinSizeT window, ValueT kz_d, ValueT max_d)
624 {
625 return (std::fabs(max_d) > 0)
626 ? std::floor(window * (1 - kz_d/max_d))
627 : window;
628 }
629
630protected:
631 WinSizeT min_window_size;
632 ValueT tolerance;
633
640 KZAExtension(WinSizeT min_window_size, ValueT tolerance)
641 : min_window_size(min_window_size), tolerance(tolerance) {}
642
652 inline WinSizeT normalize_left_win(SizeT left_win, SizeT window_center)
653 {
654 SizeT new_size = (left_win < min_window_size)
655 ? min_window_size
656 : left_win;
657
658 new_size = (new_size > window_center)
659 ? window_center
660 : new_size;
661
662 return new_size;
663 }
664
675 inline WinSizeT normalize_right_win(SizeT right_win,
676 SizeT window_center,
677 SizeT data_size)
678 {
679 SizeT new_size = (right_win < min_window_size)
680 ? min_window_size
681 : right_win;
682
683 SizeT max_right_win_size = data_size - window_center - 1;
684 new_size = (new_size > max_right_win_size)
685 ? max_right_win_size
686 : new_size;
687
688 return new_size;
689 }
690};

696template<class ValueT, std::unsigned_integral SizeT,
697 std::unsigned_integral WinSizeT>
698class KZA1D: public KZ1D<ValueT, SizeT, WinSizeT>,
699 public KZAExtension<ValueT, SizeT, WinSizeT>
700{
701public:
710 KZA1D(WinSizeT window_size, WinSizeT min_window_size,
711 ValueT tolerance, SizeT iterations)
712 : KZ1D<ValueT, SizeT, WinSizeT>(window_size, iterations),
713 KZAExtension<ValueT, SizeT, WinSizeT>(min_window_size,
714 tolerance)
715 {
716 // unlike of KZ algorithm, KZA uses (2*window_size+1) window
717 half_window = window_size;
718#ifdef KZA_TIMER
719 KZGeneric<ValueT, SizeT, WinSizeT>::timer.set_name("kza1d");
720#endif
721 }
722
729 std::vector<ValueT> operator()(const std::vector<ValueT> &data,
730 const std::vector<ValueT> &kz_result)
731 {
732 kz_diff.resize(data.size());
733 kz_derivative.resize(data.size());
734
735 calc_kz_difference(kz_result);
736 calc_kz_dirivative();
737
738 // call KZ algorithm with overrided perform_single_iteration() method
739 return KZ1D<ValueT, SizeT, WinSizeT>::operator()(data);
740 }
741
742 inline void set_window(WinSizeT window) override
743 {
744 half_window = window;
745 }
746
747private:
749 std::vector<ValueT> kz_diff;
750 ValueT max_diff;
751
753 std::vector<ValueT> kz_derivative;
754
755 using KZGeneric<ValueT, SizeT, WinSizeT>::iterations;
756
757 using KZ1D<ValueT, SizeT, WinSizeT>::half_window;
758 using KZ1D<ValueT, SizeT, WinSizeT>::data;
759 using KZ1D<ValueT, SizeT, WinSizeT>::ans;
760
761 using KZAExtension<ValueT, SizeT, WinSizeT>::min_window_size;
762 using KZAExtension<ValueT, SizeT, WinSizeT>::tolerance;
763
764
765 void perform_single_iteration(SizeT start_idx, SizeT end_idx) override
766 {
767 WinSizeT left_win_size, right_win_size;
768
769 for (SizeT time = start_idx; time <= end_idx; ++time) {
770 this->calc_adaptive_half_windows(left_win_size, right_win_size,
771 time);
772
773 ans[time] = this->average(time - left_win_size,
774 time + right_win_size);
775 }
776 }
777
787 inline void calc_adaptive_half_windows(WinSizeT &left_half_win,
788 WinSizeT &right_half_win,
789 SizeT t)
790 {
791 SizeT data_size = data.size();
792 SizeT adaptive_size = this->adaptive_window(half_window,
793 kz_diff[t], max_diff);
794
795 // derivative[t] = 0
796 if (std::fabs(kz_derivative[t]) < tolerance) {
797 left_half_win = this->normalize_left_win(adaptive_size, t);
798 right_half_win =
799 this->normalize_right_win(adaptive_size, t, data_size);
800 } else if (kz_derivative[t] < 0) {
801 left_half_win = this->normalize_left_win(adaptive_size, t);
802 right_half_win =
803 this->normalize_right_win(half_window, t, data_size);
804 } else {
805 right_half_win =
806 this->normalize_right_win(adaptive_size, t, data_size);
807 left_half_win = this->normalize_left_win(half_window, t);
808 }
809 }
810
816 inline void calc_kz_difference(const std::vector<ValueT> &y)
817 {
818 SizeT n = y.size();
819 WinSizeT q = half_window;
820 max_diff = 0;
821
822 for (SizeT i = 0; i < q; ++i) {
823 kz_diff[i] = std::fabs(y[i+q] - y[0]);
824 max_diff = std::max(max_diff, kz_diff[i]);
825 }
826 for (SizeT i = q; i < n-q; ++i) {
827 kz_diff[i] = std::fabs(y[i+q] - y[i-q]);
828 max_diff = std::max(max_diff, kz_diff[i]);
829 }
830 for (SizeT i = n-q; i < n; ++i) {
831 kz_diff[i] = std::fabs(y[n-1] - y[i-q]);
832 max_diff = std::max(max_diff, kz_diff[i]);
833 }
834 }
835
840 inline void calc_kz_dirivative()
841 {
842 SizeT n = kz_diff.size();
843
844 if (n < 2)
845 return;
846
847 for (SizeT i = 0; i < n-1; i++)
848 kz_derivative[i] = kz_diff[i+1] - kz_diff[i];
849 kz_derivative[n-1] = kz_derivative[n-2];
850 }
851};

856template<class ValueT, std::unsigned_integral SizeT,
857 std::unsigned_integral WinSizeT>
858class KZA2D: public KZ2D<ValueT, SizeT, WinSizeT>,
859 public KZAExtension<ValueT, SizeT, WinSizeT>
860{
861public:
870 KZA2D(const std::vector<WinSizeT> &window, WinSizeT min_window_size,
871 ValueT tolerance, SizeT iterations)
872 : KZ2D<ValueT, SizeT, WinSizeT>(window, iterations),
873 KZAExtension<ValueT, SizeT, WinSizeT>(min_window_size,
874 tolerance)
875 {
876 // use KZA specific window size: (2*window+1)
877 this->set_window(window);
878#ifdef KZA_TIMER
879 KZGeneric<ValueT, SizeT, WinSizeT>::timer.set_name("kza2d");
880#endif
881 }
882
889 std::vector<std::vector<ValueT>> operator()(
890 const std::vector<std::vector<ValueT>> &data,
891 const std::vector<std::vector<ValueT>> &kz_result)
892 {
893 rows = data.size();
894 cols = data[0].size();
895
896 kz_dx.resize(rows, std::vector<ValueT> (cols));
897 kz_dy.resize(rows, std::vector<ValueT> (cols));
898 x_derivative.resize(rows, std::vector<ValueT> (cols));
899 y_derivative.resize(rows, std::vector<ValueT> (cols));
900
901 calc_kz_difference(kz_result);
902 calc_kz_dirivative();
903
904 // call KZ algorithm with overrided perform_single_iteration() method
905 return KZ2D<ValueT, SizeT, WinSizeT>::operator()(data);
906 }
907
908 inline void set_window(const std::vector<WinSizeT> &window) override
909 {
910 // unlike of KZ algorithm, KZA uses (2*window+1) window
911 half_win_rows = window[0];
912 half_win_cols = window.size() > 1 ? window[1] : half_win_rows;
913 }
914
915private:
917 std::vector<std::vector<ValueT>> kz_dx;
918
920 std::vector<std::vector<ValueT>> kz_dy;
921
922 ValueT max_dx, max_dy;
923
927 std::vector<std::vector<ValueT>> x_derivative;
928
932 std::vector<std::vector<ValueT>> y_derivative;
933
934 using KZGeneric<ValueT, SizeT, WinSizeT>::iterations;
935
936 using KZ2D<ValueT, SizeT, WinSizeT>::half_win_rows;
937 using KZ2D<ValueT, SizeT, WinSizeT>::half_win_cols;
938 using KZ2D<ValueT, SizeT, WinSizeT>::data;
939 using KZ2D<ValueT, SizeT, WinSizeT>::ans;
940 using KZ2D<ValueT, SizeT, WinSizeT>::cols;
941 using KZ2D<ValueT, SizeT, WinSizeT>::rows;
942
943 using KZAExtension<ValueT, SizeT, WinSizeT>::min_window_size;
944 using KZAExtension<ValueT, SizeT, WinSizeT>::tolerance;
945
946
947 void perform_single_iteration(SizeT start_idx, SizeT end_idx) override
948 {
949 std::pair<WinSizeT, WinSizeT> rows_win;
950 std::pair<WinSizeT, WinSizeT> cols_win;
951
952 for (SizeT row = start_idx; row <= end_idx; ++row) {
953 for (SizeT col = 0; col < cols; ++col) {
954 rows_win = this->adaptive_window_halves(
955 row, half_win_rows, rows,
956 kz_dy[row][col], max_dy,
957 y_derivative[row][col]
958 );
959 cols_win = this->adaptive_window_halves(
960 col, half_win_cols, cols,
961 kz_dx[row][col], max_dx,
962 x_derivative[row][col]
963 );
964
965 // check that area is not zero
966 ans[row][col] = (rows_win.second + rows_win.first +
967 cols_win.second + cols_win.first)
968
969 ? this->average(row - rows_win.first,
970 row + rows_win.second,
971 col - cols_win.first,
972 col + cols_win.second)
973

992 SizeT win_center, WinSizeT window,
993 SizeT data_size,
994 ValueT kz_d, ValueT max_d,
995 ValueT derivative)
996 {
997 std::pair<WinSizeT, WinSizeT> adaptive_win;

1003 this->normalize_left_win(adaptive, win_center);

1009 this->normalize_left_win(adaptive, win_center);
1010 adaptive_win.second =

1012 } else {
1013 adaptive_win.second =
1014 this->normalize_right_win(adaptive, win_center, data_size);
1015 adaptive_win.first =
1016 this->normalize_left_win(window, win_center);
1017 }

1026 inline void calc_kz_difference(
1027 const std::vector<std::vector<ValueT>> &kz)
1028 {
1029 WinSizeT q1 = half_win_rows;
1030 WinSizeT q2 = half_win_cols;
1031
1032 max_dx = 0;
1033 max_dy = 0;
1034
1035 for (SizeT i = 0; i < rows; ++i) { // y coord
1036 for (SizeT j = 0; j < cols; ++j) { // x coord
1037 SizeT row_tail = this->window_left_bound(i, q1);
1038 SizeT col_tail = this->window_left_bound(j, q2);
1039 SizeT row_head = this->window_right_bound(i, q1, rows);
1040 SizeT col_head = this->window_right_bound(j, q2, cols);
1041
1042 kz_dx[i][j] = std::fabs(kz[i][col_head] - kz[i][col_tail]);
1043 kz_dy[i][j] = std::fabs(kz[row_head][j] - kz[row_tail][j]);
1044
1045 max_dx = std::max(max_dx, kz_dx[i][j]);
1046 max_dy = std::max(max_dy, kz_dy[i][j]);
1047 }
1048 }
1049 }

1055 inline void calc_kz_dirivative()
1056 {
1057 /* d'(t) = d(i+1)-d(i) */
1058
1059 for (SizeT i = 0; i < rows-1; ++i) {
1060 for (SizeT j = 0; j < cols-1; ++j) {
1061 x_derivative[i][j] = kz_dx[i][j+1] - kz_dx[i][j];
1062 y_derivative[i][j] = kz_dy[i+1][j] - kz_dy[i][j];
1063 }
1064 }
1065 for (SizeT i = 0; i < rows-1; ++i) {
1066 x_derivative[i][cols-1] = x_derivative[i][cols-2];
1067 y_derivative[i][cols-1] = kz_dy[i+1][cols-1] - kz_dy[i][cols-1];
1068 }
1069 for (SizeT j = 0; j < cols-1; ++j) {
1070 y_derivative[rows-1][j] = y_derivative[rows-2][j];
1071 x_derivative[rows-1][j] = kz_dx[rows-1][j+1] - kz_dx[rows-1][j];
1072 }
1073 }
1074};
